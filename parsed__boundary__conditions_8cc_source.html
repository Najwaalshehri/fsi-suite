246  template <int spacedim>
247  void
248  ParsedBoundaryConditions<spacedim>::check_consistency() const
249  {
250  n_boundary_conditions = ids.size();
251  AssertThrow(n_boundary_conditions == bc_type.size(),
252  ExcMessage("The number of boundary ids must be equal to the "
253  "number of boundary condition types."));
254  AssertThrow(n_boundary_conditions == expressions.size(),
255  ExcMessage("The number of boundary ids must be equal to the "
256  "number of expressions."));
257  AssertThrow(n_boundary_conditions == selected_components.size(),
258  ExcMessage("The number of boundary ids "
259  "must be equal to the number "
260  "of selected components."));
261 
262  std::set<types::boundary_id> all_ids;
263  unsigned int n_ids = 0;
264  for (const auto &id : ids)
265  {
266  all_ids.insert(id.begin(), id.end());
267  n_ids += id.size();
268  }
269 
270  // Special meaning of boundary id numbers::invalid_boundary_id:
271  if (all_ids.find(numbers::invalid_boundary_id) != all_ids.end())
272  {
273  AssertThrow(all_ids.size() == 1,
274  ExcMessage("If you use the boundary id -1, then no other "
275  "boundary ids can be specified"));
276  AssertThrow(n_boundary_conditions == 1,
277  ExcMessage("Only one BoundaryCondition can be specified "
278  "with the special boundary id -1"));
279  }
280  else
281  {
282  AssertThrow(all_ids.size() == n_ids,
283  ExcMessage("You specified the same "
284  " boundary id more than once "
285  "in two different boundary conditions"));
286  // We check consistency with the triangulation
287  if (grid_info.n_active_cells > 0)
288  {
289  AssertThrow(all_ids.size() == grid_info.boundary_ids.size(),
290  ExcMessage("The number of boundary ids specified in "
291  "the input file does not match the number "
292  "of boundary ids in the triangulation"));
293  }
294  }
295 
296  // Now check that the types are valid in this dimension
297  AssertDimension(n_boundary_conditions, types.size());
298  AssertDimension(n_boundary_conditions, masks.size());
299  for (unsigned int i = 0; i < n_boundary_conditions; ++i)
300  {
301  if (types[i] == Components::Type::vector ||
302  types[i] == Components::Type::normal ||
303  types[i] == Components::Type::tangential)
304  {
305  AssertThrow(masks[i].n_selected_components() == spacedim,
306  ExcDimensionMismatch(masks[i].n_selected_components(),
307  spacedim));
308  }
309  if (types[i] == Components::Type::normal ||
310  types[i] == Components::Type::tangential)
311  {
312  AssertThrow(functions[i]->n_components == spacedim,
313  ExcDimensionMismatch(functions[i]->n_components,
314  spacedim));
315  }
316  else
317  {
318  // In all other cases, it is the mask that determines the number
319  // of of components, but the function needs to be of the correct
320  // dimension
321  AssertThrow(functions[i]->n_components == n_components,
322  ExcDimensionMismatch(functions[i]->n_components,
323  n_components));
324  }
325  }
326  }
