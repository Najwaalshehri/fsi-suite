153template <int dim, int spacedim>
154void
155ParsedGridGenerator<dim, spacedim>::write(
156 const dealii::Triangulation<dim, spacedim> &tria,
157 const std::string & filename) const
158{
159 const std::string outname = filename != "" ? filename : output_file_name;
160 if (outname != "")
161 {
162 const auto ext = boost::algorithm::to_lower_copy(
163 outname.substr(outname.find_last_of('.') + 1));
164
165 GridOut go;
166 std::ofstream out(outname);
167 AssertThrow(out, ExcIO());
168
169 go.set_flags(GridOutFlags::Msh(true, true));
170 go.set_flags(GridOutFlags::Ucd(false, true, true));
171 go.set_flags(GridOutFlags::Vtu(true));
172
173 if (ext == "vtk")
174 go.write_vtk(tria, out);
175 else if (ext == "msh")
176 go.write_msh(tria, out);
177 else if (ext == "vtu")
178 go.write_vtu(tria, out);
179 else if (ext == "ucd" || ext == "inp")
180 go.write_ucd(tria, out);
181 else if (ext == "vtu")
182 go.write_vtu(tria, out);
183 else if (ext == "ar")
184 {
185 boost::archive::text_oarchive oa(out);
186 tria.save(oa, 0);
187 }
188 else if (ext == "bin")
189 {
190 boost::archive::binary_oarchive oa(out);
191 tria.save(oa, 0);
192 }
193 else
194 Assert(false, ExcNotImplemented());
195 out.close();
196 }
197}
