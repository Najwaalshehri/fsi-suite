23  , fe_name(fe_name)
24  {

27  unsigned int j = 0;
28  for (unsigned int i = 0; i < component_names.size(); ++i)
29  {
30  if ((i > 0) && (component_names[i - 1] != component_names[i]))
31  j++;
32  component_blocks[i] = j;
33  block_names[j] = component_names[i];
34  }
35  block_names.resize(j + 1);
36

38  this->fe_name);
39  enter_my_subsection(this->prm);
40  this->prm.add_action(
41  "Finite element space (" + joint_component_names + ")",
42  [&](const std::string &value) {
43  fe = FETools::get_fe_by_name<dim, spacedim>(value);
44  // Check that the number of components is correct
45  Assert(fe->n_components() == component_names.size(),
46  ExcMessage("The number of components in the finite element "
