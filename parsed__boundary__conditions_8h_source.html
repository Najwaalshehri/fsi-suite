290  template <int dim>
291  void
292  ParsedBoundaryConditions<spacedim>::apply_boundary_conditions(
293  const dealii::Mapping<dim, spacedim> & mapping,
294  const dealii::DoFHandler<dim, spacedim> &dof_handler,
295  dealii::AffineConstraints<double> & constraints,
296  const bool & ignore_unsopported) const
297  {
298  // Take care of boundary conditions that don't need anything else than
299  // the constraints.
300  for (unsigned int i = 0; i < n_boundary_conditions; ++i)
301  {
302  const auto &boundary_ids = ids[i];
303  const auto &bc = bc_type[i];
304  const auto &function = functions[i];
305  const auto &mask = masks[i];
306  const auto &type = types[i];
307  std::map<dealii::types::boundary_id, const dealii::Function<spacedim> *>
308  fmap;
309 
310  if (boundary_ids.find(dealii::numbers::internal_face_boundary_id) !=
311  boundary_ids.end())
312  {
313  const auto all_ids =
314  dof_handler.get_triangulation().get_boundary_ids();
315  for (const auto &id : all_ids)
316  fmap[id] = function.get();
317  }
318  else
319  for (const auto &id : boundary_ids)
320  fmap[id] = function.get();
321 
322  // In this function, we only do Dirichlet boundary conditions.
323  switch (bc)
324  {
325  case BoundaryConditionType::dirichlet:
326  switch (type)
327  {
328  case Components::Type::normal:
329  if constexpr (dim == spacedim && dim > 1)
330  dealii::VectorTools::
331  compute_nonzero_normal_flux_constraints(
332  dof_handler,
333  mask.first_selected_component(),
334  boundary_ids,
335  fmap,
336  constraints,
337  mapping);
338  else
339  AssertThrow(false,
340  dealii::ExcMessage(
341  "Cannot use normal "
342  "flux boundary conditions "
343  "for this dim and spacedim"));
344  break;
345  case Components::Type::tangential:
346  if constexpr (dim == spacedim && dim > 1)
347  dealii::VectorTools::
348  compute_nonzero_tangential_flux_constraints(
349  dof_handler,

351  boundary_ids,
352  fmap,
353  constraints,
354  mapping);
355  else

361  break;

364  mapping, dof_handler, fmap, constraints, mask);
365  break;
