332  const bool & ignore_unsopported) const

337  {

343  std::map<dealii::types::boundary_id, const dealii::Function<spacedim> *>
344  fmap;
345 
346  if (boundary_ids.find(dealii::numbers::internal_face_boundary_id) !=
347  boundary_ids.end())
348  {
349  const auto all_ids =
350  dof_handler.get_triangulation().get_boundary_ids();
351  for (const auto &id : all_ids)
352  fmap[id] = function.get();
353  }
354  else
355  for (const auto &id : boundary_ids)
356  fmap[id] = function.get();
357 
358  // In this function, we only do Dirichlet boundary conditions.
359  if (bc == BoundaryConditionType::dirichlet)
360  {
361  if (type == Components::Type::normal)
362  {
363  dealii::VectorTools::compute_nonzero_normal_flux_constraints(
364  dof_handler,
365  mask.first_selected_component(),
366  boundary_ids,
367  fmap,
368  constraints,
369  mapping);
370  }
371  else if (type == Components::Type::tangential)
372  {
373  dealii::VectorTools::
