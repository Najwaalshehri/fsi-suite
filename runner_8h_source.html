111 
112  // Either as -p prm_file or as first positional argument after all options

114  cli(1, input_parameter_file) >> input_parameter_file;
115 
116  cli({"d", "dim"}, 2) >> dim;
117  cli({"s", "spacedim"}, 2) >> spacedim;
118 
119  if (input_parameter_file != "")

121  output_parameter_file = "used_" + input_parameter_file;

123  else if (dim == spacedim)

125  output_parameter_file =
126  "used_" + exename + "_" + std::to_string(dim) + "d.prm";
127  }
128  else
129  {
130  output_parameter_file = "used_" + exename + "_" + std::to_string(dim) +
131  "d_" + std::to_string(spacedim) + "d.prm";
132  }
133 
134  // If you want to overwrite the output parameter file, use the -o option
135  cli({"o", "output_prm_file"}, output_parameter_file) >> output_parameter_file;
136 
137  deallog << "Will run in dimension " << dim << " and spacedimemsion "
138  << spacedim << std::endl
139  << "Input parameter file: " << input_parameter_file << std::endl
140  << "Output parameter file: " << output_parameter_file << std::endl;
141 
142  return std::make_tuple(dim,
143  spacedim,
144  input_parameter_file,
145  output_parameter_file);
