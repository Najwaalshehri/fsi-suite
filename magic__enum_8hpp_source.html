279 template <typename L, typename R>
280 constexpr bool cmp_less(L lhs, R rhs) noexcept {
281  static_assert(std::is_integral_v<L> && std::is_integral_v<R>, "magic_enum::detail::cmp_less requires integral type.");
282 
283  if constexpr (std::is_signed_v<L> == std::is_signed_v<R>) {
284  // If same signedness (both signed or both unsigned).
285  return lhs < rhs;
286  } else if constexpr (std::is_same_v<L, bool>) { // bool special case due to msvc's C4804, C4018
287  return static_cast<R>(lhs) < rhs;
288  } else if constexpr (std::is_same_v<R, bool>) { // bool special case due to msvc's C4804, C4018
289  return lhs < static_cast<L>(rhs);
290  } else if constexpr (std::is_signed_v<R>) {
291  // If 'right' is negative, then result is 'false', otherwise cast & compare.
292  return rhs > 0 && lhs < static_cast<std::make_unsigned_t<R>>(rhs);
293  } else {
294  // If 'left' is negative, then result is 'true', otherwise cast & compare.
295  return lhs < 0 || static_cast<std::make_unsigned_t<L>>(lhs) < rhs;
296  }
297 }
